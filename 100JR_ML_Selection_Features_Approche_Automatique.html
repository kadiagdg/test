import scipy.stats as ss

def cramers_v(x, y):
    confusion_matrix = pd.crosstab(x, y)
    chi2 = ss.chi2_contingency(confusion_matrix)[0]
    n = confusion_matrix.sum().sum()
    phi2 = chi2 / n
    r, k = confusion_matrix.shape
    phi2corr = max(0, phi2 - ((k-1)*(r-1))/(n-1))
    rcorr = r - ((r-1)**2)/(n-1)
    kcorr = k - ((k-1)**2)/(n-1)
    return np.sqrt(phi2corr / min((kcorr-1), (rcorr-1)))

def tchuprow_t(x, y):
    confusion_matrix = pd.crosstab(x, y)
    chi2 = ss.chi2_contingency(confusion_matrix)[0]
    n = confusion_matrix.sum().sum()
    phi2 = chi2 / n
    r, k = confusion_matrix.shape
    return np.sqrt(phi2 / (np.sqrt((k - 1) * (r - 1))))

Sélection basée sur l'information value¶

L'Information Value (IV) est une mesure utilise pour valuer la puissance prdictive d'une variable catgorielle dans un modle de rgression binaire. L'IV est bas sur le concept de Weight of Evidence (WOE), qui est une mesure logarithmique des proportions d'vnements et de non-vnements pour chaque catgorie d'une variable.

Calcul de l'Information Value¶

L'Information Value pour une variable est calcule en utilisant la formule suivante :

$$IV = \sum_{i=1}^n (EventDist_i - NonEventDist_i) * WOE_i$$

o $n$ est le nombre de catgories pour la variable, $EventDist_i$ est la proportion d'vnements pour la catgorie $i$, $NonEventDist_i$ est la proportion de non-vnements pour la catgorie $i$, et $WOE_i$ est le Weight of Evidence pour la catgorie $i$.

Le Weight of Evidence pour une catgorie est calcul en utilisant la formule suivante :

$$WOE_i = \ln \frac{EventDist_i}{NonEventDist_i}$$

Interprétation de l'Information Value¶

L'Information Value peut tre interprt comme suit :

  • IV < 0.02 : La variable a trs peu de pouvoir prdictif.
  • 0.02 <= IV < 0.1 : La variable a un faible pouvoir prdictif.
  • 0.1 <= IV < 0.3 : La variable a un pouvoir prdictif moyen.
  • 0.3 <= IV < 0.5 : La variable a un pouvoir prdictif fort.
  • IV >= 0.5 : La variable a un pouvoir prdictif trs fort, mais cela peut indiquer une fuite d'informations (data leakage) ou un problme avec les donnes.

# Calculate information value
def calc_iv(df, feature, target, pr=0, epsilon=1e-6):

    lst = []

    for i in range(df[feature].nunique()):
        val = list(df[feature].unique())[i]
        lst.append([feature, val, df[df[feature] == val].count()[feature], df[(df[feature] == val) & (df[target] == 1)].count()[feature]])

    data = pd.DataFrame(lst, columns=['Variable', 'Value', 'All', 'Bad'])
    data = data[data['Bad'] > 0]

    data['Share'] = data['All'] / data['All'].sum()
    data['Bad Rate'] = data['Bad'] / data['All']
    data['Distribution Good'] = (data['All'] - data['Bad']) / (data['All'].sum() - data['Bad'].sum())
    data['Distribution Bad'] = data['Bad'] / data['Bad'].sum()
    data['WoE'] = np.log((data['Distribution Good'] + epsilon) / (data['Distribution Bad'] + epsilon))
    data['IV'] = (data['WoE'] * (data['Distribution Good'] - data['Distribution Bad'])).sum()

    data = data.sort_values(by=['Variable', 'Value'], ascending=True)

    if pr == 1:
        print(data)

    return data['IV'].values[0]

def iv_woe(data, target, bins=10, show_woe=False):
    
    #Empty Dataframe
    newDF,woeDF = pd.DataFrame(), pd.DataFrame()
    
    #Extract Column Names
    cols = data.columns
    
    #Run WOE and IV on all the independent variables
    for ivars in cols[~cols.isin([target])]:
        if (data[ivars].dtype.kind in 'bifc') and (len(np.unique(data[ivars]))>10):
            binned_x = pd.qcut(data[ivars], bins,  duplicates='drop')
            d0 = pd.DataFrame({'x': binned_x, 'y': data[target]})
        else:
            d0 = pd.DataFrame({'x': data[ivars], 'y': data[target]})

        
        # Calculate the number of events in each group (bin)
        d = d0.groupby("x", as_index=False).agg({"y": ["count", "sum"]})
        d.columns = ['Cutoff', 'N', 'Events']
        
        # Calculate % of events in each group.
        d['% of Events'] = np.maximum(d['Events'], 0.5) / d['Events'].sum()

        # Calculate the non events in each group.
        d['Non-Events'] = d['N'] - d['Events']
        # Calculate % of non events in each group.
        d['% of Non-Events'] = np.maximum(d['Non-Events'], 0.5) / d['Non-Events'].sum()

        # Calculate WOE by taking natural log of division of % of non-events and % of events
        d['WoE'] = np.log(d['% of Events']/d['% of Non-Events'])
        d['IV'] = d['WoE'] * (d['% of Events'] - d['% of Non-Events'])
        d.insert(loc=0, column='Variable', value=ivars)
        print("Information value of " + ivars + " is " + str(round(d['IV'].sum(),6)))
        temp =pd.DataFrame({"Variable" : [ivars], "IV" : [d['IV'].sum()]}, columns = ["Variable", "IV"])
        newDF=pd.concat([newDF,temp], axis=0)
        woeDF=pd.concat([woeDF,d], axis=0)

        #Show WOE Table
        if show_woe == True:
            print(d)
    return newDF, woeDF

def information_value(x, y, epsilon=1e-6):
    df = pd.DataFrame({'feature': x, 'target': y})
    df['bins'] = pd.cut(df['feature'], bins=pd.Series(np.unique(x)).nunique())
    grouped = df.groupby('bins')['target'].agg(['count', 'sum'])
    grouped['non_event'] = grouped['count'] - grouped['sum']
    grouped['event_dist'] = (grouped['sum'] + epsilon) / (grouped['sum'].sum() + epsilon)
    grouped['non_event_dist'] = (grouped['non_event'] + epsilon) / (grouped['non_event'].sum() + epsilon)
    grouped['woe'] = np.log(grouped['event_dist'] / grouped['non_event_dist'])
    grouped['iv'] = (grouped['event_dist'] - grouped['non_event_dist']) * grouped['woe']
    return grouped['iv'].sum()

Liste des méthodes de filtrage¶

Les mthodes de slection de variables par filtrage (filter methods) valuent la pertinence des variables en fonction de leurs proprits statistiques, sans tenir compte du modle de machine learning spcifique. Voici quelques mthodes de filtrage couramment utilises pour la slection des variables :

Corrlation de Pearson : Cette mthode mesure la corrlation linaire entre chaque variable et la variable cible. Les variables ayant une corrlation leve (positive ou ngative) sont considres comme pertinentes.

Chi-carr (Chi-squared) : Le test du chi-carr est utilis pour dterminer si une relation significative existe entre deux variables catgorielles. Il est couramment utilis pour la slection de caractristiques lorsque les variables d'entre et la variable cible sont toutes deux catgorielles.

Coefficient de corrlation de Cramer : Le coefficient de corrlation de Cramer est une mesure de la force de l'association entre deux variables catgorielles. Il peut tre utilis pour mesurer la relation entre deux variables catgorielles, comme la taille et la couleur, ou la marque et la couleur d'un produit. Le coefficient de corrlation de Cramer varie de 0 1, o 0 indique une absence d'association et 1 indique une association parfaite.

ANOVA (Analyse de la variance) : L'ANOVA est utilise pour dterminer si les moyennes de plusieurs groupes sont gales. Dans la slection de caractristiques, l'ANOVA est souvent utilise pour comparer les moyennes des groupes forms en fonction de la variable cible pour chaque variable d'entre.

Test de Kruskal-Wallis : Le test de Kruskal-Wallis est une alternative non paramtrique l'ANOVA qui teste si plusieurs groupes ont la mme distribution. Il est utile pour la slection de caractristiques lorsque les donnes ne satisfont pas aux hypothses de l'ANOVA (par exemple, lorsque les donnes ne sont pas normalement distribues).

Variance: La slection de variables base sur la variance consiste choisir les variables dont la variance est suprieure un certain seuil. Les variables ayant une faible variance sont considres comme moins informatives car elles ne varient pas beaucoup entre les observations.

Corrlation de Pearson : Cette mthode mesure la corrlation linaire entre deux variables continues. Elle peut tre utilise pour dterminer si deux variables sont lies et dans quelle mesure elles varient ensemble. La corrlation de Pearson peut tre utilise pour valuer la relation entre une variable indpendante et une variable dpendante.

Test de Wilcoxon : Le test de Wilcoxon est une mthode non paramtrique utilise pour tester si deux chantillons indpendants ont la mme distribution. Il peut tre utilis pour comparer les diffrences entre les variables continues de deux groupes. Le test de Wilcoxon est souvent utilis dans les tudes cliniques pour valuer les effets de traitements ou de thrapies sur une variable mesure chez des patients.

Information Value (IV) : L'Information Value est une mesure utilise pour valuer la capacit d'une variable prdire la variable cible dans le contexte des modles de rgression logistique. Elle est base sur les rapports de cotes (Odds Ratios) et le concept de Weight of Evidence (WOE).

Recursive Feature Elimination¶

La mthode Recursive Feature Elimination (RFE) est une technique de slection de caractristiques base sur la rcursivit, qui vise slectionner les caractristiques les plus pertinentes pour un modle d'apprentissage automatique. L'ide principale de RFE est de construire rptitivement un modle et d'liminer les caractristiques les moins importantes, jusqu' ce qu'un nombre souhait de caractristiques soit atteint.

Voici comment fonctionne l'algorithme RFE :

  1. Entranez un modle sur l'ensemble des caractristiques et calculez l'importance de chaque caractristique (par exemple, les coefficients d'un modle linaire ou les importances des caractristiques d'un arbre de dcision).
  2. Supprimez la caractristique la moins importante du jeu de donnes.
  3. Rptez les tapes 1 et 2 jusqu' ce que toutes les caractristiques aient t limines.
  4. valuez les performances du modle pour chaque sous-ensemble de caractristiques limines et choisissez le sous-ensemble qui maximise les performances (par exemple, en utilisant la validation croise).

RFE peut tre utilis avec diffrents modles d'apprentissage automatique et est implment dans la bibliothque scikit-learn avec la classe RFECV. Cette classe effectue galement une validation croise pour dterminer le nombre optimal de caractristiques conserver.

Avantages de RFE¶

  • RFE prend en compte les relations entre les caractristiques, contrairement aux mthodes de filtrage.
  • RFE est compatible avec diffrents modles d'apprentissage automatique et est facile utiliser avec la bibliothque scikit-learn.
  • RFE value les performances du modle chaque tape, ce qui permet de choisir le nombre optimal de caractristiques.

Inconvénients de RFE¶

  • RFE peut tre lent, surtout pour les jeux de donnes avec un grand nombre de caractristiques, car il doit construire un modle pour chaque sous-ensemble de caractristiques.
  • Les performances de RFE dpendent fortement du modle d'apprentissage automatique utilis pour valuer l'importance des caractristiques.
  • RFE peut ne pas tre efficace si le modle d'apprentissage automatique utilis ne reflte pas l'importance des caractristiques de manire prcise.
